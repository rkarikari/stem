Radio Control via CAT Interface

üì° CAT Communication Architecture

Key Components: The computer runs Python with PySerial, connecting via USB to either a direct radio port or through a CAT interface adapter. Commands use manufacturer-specific protocols (Icom CI-V, Yaesu CAT, etc.).

üîå Serial Communication Protocol

Control Transceiver from Python

Use PySerial to send CAT commands to radio

import serial import time class RadioController: def __init__(self, port, baudrate=19200): """Initialize connection to radio""" self.ser = serial.Serial( port=port, baudrate=baudrate, bytesize=serial.EIGHTBITS, parity=serial.PARITY_NONE, stopbits=serial.STOPBITS_ONE, timeout=1 ) time.sleep(0.1) # Allow connection to stabilize def set_frequency(self, freq_hz): """Set VFO frequency (Icom CI-V format)""" # Convert frequency to BCD format freq_str = f"{freq_hz:010d}" bcd_bytes = [] for i in range(0, 10, 2): bcd_bytes.append(int(freq_str[i+1] + freq_str[i], 16)) # Icom CI-V command: FE FE [radio_addr] [pc_addr] 05 [freq_bcd] FD cmd = [0xFE, 0xFE, 0x94, 0xE0, 0x05] + bcd_bytes + [0xFD] self.ser.write(bytes(cmd)) time.sleep(0.05) # Read response response = self.ser.read(100) return 0xFB in response # FB = ACK def get_frequency(self): """Read current VFO frequency""" # Send read frequency command cmd = [0xFE, 0xFE, 0x94, 0xE0, 0x03, 0xFD] self.ser.write(bytes(cmd)) time.sleep(0.05) response = self.ser.read(100) if len(response) >= 11: # Parse BCD frequency from response freq_bcd = response[5:10] freq_str = "" for byte in freq_bcd: freq_str = f"{byte:02X}" + freq_str return int(freq_str) return None def set_mode(self, mode): """Set operating mode""" modes = {'LSB': 0x00, 'USB': 0x01, 'AM': 0x02, 'CW': 0x03, 'FM': 0x05} if mode not in modes: return False cmd = [0xFE, 0xFE, 0x94, 0xE0, 0x06, modes[mode], 0xFD] self.ser.write(bytes(cmd)) time.sleep(0.05) response = self.ser.read(100) return 0xFB in response def close(self): """Close serial connection""" self.ser.close() # Example usage radio = RadioController('/dev/ttyUSB0') # Set frequency to 146.520 MHz (2m calling frequency) if radio.set_frequency(146520000): print("Frequency set successfully") # Set mode to FM radio.set_mode('FM') # Read back frequency current_freq = radio.get_frequency() print(f"Current frequency: {current_freq/1e6:.4f} MHz") radio.close()

System Flow: TLE (Two-Line Element) data provides orbital parameters, SGP4 algorithm calculates real-time position relative to ground station, outputs drive antenna rotors and apply Doppler correction to radio frequency.

from skyfield.api import load, wgs84, EarthSatellite from datetime import datetime, timezone class SatelliteTracker: def __init__(self, tle_line1, tle_line2, observer_lat, observer_lon, observer_alt): """Initialize satellite tracker""" self.ts = load.timescale() self.satellite = EarthSatellite(tle_line1, tle_line2, 'Satellite', self.ts) self.observer = wgs84.latlon(observer_lat, observer_lon, observer_alt) def get_position(self, dt=None): """Get satellite position at given time""" if dt is None: dt = datetime.now(timezone.utc) t = self.ts.from_datetime(dt) difference = self.satellite - self.observer topocentric = difference.at(t) alt, az, distance = topocentric.altaz() return { 'altitude': alt.degrees, 'azimuth': az.degrees, 'distance_km': distance.km, 'timestamp': dt } def predict_pass(self, start_time, duration_hours=24, min_elevation=10): """Predict upcoming passes""" t0 = self.ts.from_datetime(start_time) t1 = self.ts.from_datetime( start_time + timedelta(hours=duration_hours) ) # Find passes above horizon t, events = self.satellite.find_events( self.observer, t0, t1, altitude_degrees=min_elevation ) passes = [] for ti, event in zip(t, events): if event == 0: # AOS aos_time = ti.utc_datetime() elif event == 1: # Maximum elevation tca_time = ti.utc_datetime() position = self.get_position(tca_time) max_el = position['altitude'] elif event == 2: # LOS los_time = ti.utc_datetime() duration = (los_time - aos_time).total_seconds() passes.append({ 'aos': aos_time, 'tca': tca_time, 'los': los_time, 'max_elevation': max_el, 'duration_minutes': duration / 60 }) return passes # Example: Track AO-91 tle1 = "1 43017U 17073E 25345.78916667 .00000342 00000-0 43210-4 0 9998" tle2 = "2 43017 97.7408 273.6829 0008372 58.9347 301.2947 14.81641075326482" tracker = SatelliteTracker( tle1, tle2, observer_lat=5.603, # Accra, Ghana observer_lon=-0.187, observer_alt=61 ) # Get current position pos = tracker.get_position() print(f"Azimuth: {pos['azimuth']:.1f}¬∞") print(f"Elevation: {pos['altitude']:.1f}¬∞") print(f"Distance: {pos['distance_km']:.1f} km") # Predict next 7 days of passes passes = tracker.predict_pass(datetime.now(timezone.utc), duration_hours=168) print(f"\nFound {len(passes)} passes:") for i, p in enumerate(passes[:5], 1): print(f"{i}. #include <complex> #include <vector> #include <cmath> #include <iostream> class FFTProcessor { private: std::vector<std::complex<double>> twiddle_factors; size_t fft_size; void compute_twiddle_factors() { twiddle_factors.resize(fft_size / 2); for (size_t k = 0; k < fft_size / 2; ++k) { double angle = -2.0 * M_PI * k / fft_size; twiddle_factors[k] = std::complex<double>( std::cos(angle), std::sin(angle) ); } } public: FFTProcessor(size_t size) : fft_size(size) { compute_twiddle_factors(); } void fft(std::vector<std::complex<double>>& data) { // Cooley-Tukey FFT algorithm size_t n = data.size(); // Bit-reversal permutation for (size_t i = 0, j = 0; i < n; ++i) { if (j > i) { std::swap(data[i], data[j]); } size_t m = n / 2; while (m >= 1 && j >= m) { j -= m; m /= 2; } j += m; } // Danielson-Lanczos section for (size_t step = 2; step <= n; step *= 2) { size_t half_step = step / 2; size_t twiddle_step = n / step; for (size_t k = 0; k < n; k += step) { for (size_t j = 0; j < half_step; ++j) { auto t = twiddle_factors[j * twiddle_step] * data[k + j + half_step]; auto u = data[k + j]; data[k + j] = u + t; data[k + j + half_step] = u - t; } } } } std::vector<double> compute_power_spectrum( const std::vector<double>& samples ) { // Convert real samples to complex std::vector<std::complex<double>> complex_data(fft_size); for (size_t i = 0; i < samples.size() && i < fft_size; ++i) { complex_data[i] = std::complex<double>(samples[i], 0.0); } // Perform FFT fft(complex_data); // Compute power spectrum std::vector<double> power_spectrum(fft_size); for (size_t i = 0; i < fft_size; ++i) { power_spectrum[i] = std::norm(complex_data[i]); } return power_spectrum; } }; // Usage example int main() { const size_t FFT_SIZE = 1024; FFTProcessor fft(FFT_SIZE); // Generate test signal: 1 kHz sine wave std::vector<double> samples(FFT_SIZE); double sample_rate = 8000.0; // 8 kHz double frequency = 1000.0; // 1 kHz for (size_t i = 0; i < FFT_SIZE; ++i) { double t = i / sample_rate; samples[i] = std::sin(2.0 * M_PI * frequency * t); } // Compute spectrum auto spectrum = fft.compute_power_spectrum(samples); // Find peak frequency auto max_iter = std::max_element( spectrum.begin(), spectrum.begin() + FFT_SIZE/2 ); size_t peak_bin = std::distance(spectrum.begin(), max_iter); double peak_freq = peak_bin * sample_rate / FFT_SIZE; std::cout << "Detected peak at " << peak_freq << " Hz\n"; return 0; }
